View w_ticket_status_f
Cube tickets X dates X status
CREATE VIEW [w_ticket_status_f] AS SELECT CASE WHEN f.close_dt_wid == 0 THEN o.calendar_date ELSE c.calendar_date END AS calendar_date, CASE WHEN f.close_dt_wid == 0 THEN o.calendar_year ELSE c.calendar_year END AS calendar_year, CASE WHEN f.close_dt_wid == 0 THEN o.calendar_month ELSE c.calendar_month END AS calendar_month, CASE WHEN f.close_dt_wid == 0 THEN o.calendar_week ELSE c.calendar_week END AS calendar_week, f.ticket_number, f.priority, t.type_name, CASE WHEN s.is_open_status = 'Y' THEN 1 ELSE 0 END AS open, CASE WHEN f.close_dt_wid <> 0 AND s.is_open_status = 'N' THEN 1 ELSE 0 END AS closed, CASE WHEN s.is_open_status = 'Y' THEN 'Open' WHEN f.close_dt_wid <> 0 AND s.is_open_status = 'N' THEN 'Closed' ELSE 'Cancelled' END AS open_closed, o.calendar_date AS open_date, c.calendar_date AS close_date, r.calendar_date AS resolved_date, CASE WHEN f.resolved_dt_wid <> 0 THEN 1 ELSE 0 END as resolved, f.status, s.status_name, b.row_wid AS board, b.name AS board_name, s.is_open_status, f.customer_name FROM w_ticket_f f INNER JOIN w_date_d o ON f.open_dt_wid = o.row_wid INNER JOIN w_date_d c ON f.close_dt_wid = c.row_wid INNER JOIN w_date_d r ON f.resolved_dt_wid = r.row_wid INNER JOIN w_type_d t ON f.type_wid = t.row_wid INNER JOIN w_status_d s ON f.status = s.row_wid INNER JOIN w_board_d b ON f.board_wid = b.row_wid


View w_ticket_priority_f
CREATE VIEW [w_ticket_priority_f] AS select p.priority_name, count(1) as cnt from w_ticket_status_f f inner join w_priority_d p on f.priority = p.row_wid where f.open = 1 group by f.priority


View w_ticket_date_f
Aggregation fact table that group open and closed tickets by: day, week, month and year.
CREATE VIEW [w_ticket_date_f] AS select 'Today' as period, sum(open) as open, sum(closed) as closed from w_ticket_status_f where calendar_date = date('now') and (open = 1 or closed = 1) group by calendar_date union all select 'This Week' as period, sum(open) as open, sum(closed) as closed from w_ticket_status_f where cast(calendar_week as text) = strftime('%W', 'now') and cast(calendar_year as text) = strftime('%Y', 'now') and (open = 1 or closed = 1) group by calendar_week, calendar_year union all select 'This Month' as period, sum(open) as open, sum(closed) as closed from w_ticket_status_f where calendar_month = cast(strftime('%m', 'now') as int) and cast(calendar_year as text) = strftime('%Y', 'now') and (open = 1 or closed = 1) group by calendar_month, calendar_year union all select 'This Year' as period, sum(open) as open, sum(closed) as closed from w_ticket_status_f where cast(calendar_year as text) = strftime('%Y','now') and (open = 1 or closed = 1) group by calendar_year


View w_sensor_clarity_stats_f
CREATE VIEW w_sensor_clarity_stats_f as select 'Clarity' as service_name, group_name, device_name, sensor_name, uptime, first_uptime_date, case when (julianday(date('now')) - julianday(last_downtime_date)) > 30 then 100 else uptime end uptime_last_30 from w_sensor_f where device_name in ('VOIP-PBX01','THCPBX01.thameshospicecare.local')


View w_sensor_vcloud_stats_f
CREATE VIEW w_sensor_vcloud_stats_f as select 'vCloud' as service_name, group_name, device_name, sensor_name, uptime, first_uptime_date, case when (julianday(date('now')) - julianday(last_downtime_date)) > 30 then 100 else uptime end uptime_last_30 from w_sensor_f where device_name like '%vCloud%'
